122 void ScalarFaspSolver::AssembleMat(const vector<vector<USI>>& colId,
123  const vector<vector<OCP_DBL>>& val,
124  const OCP_USI& dim,
125  const USI& blockDim,
126  vector<OCP_DBL>& rhs,
127  vector<OCP_DBL>& u)
128 {
129  // b & x
130  b.row = dim;
131  b.val = rhs.data();
132  x.row = dim;
133  x.val = u.data();
134  // A
135  OCP_USI nnz = 0;
136  for (OCP_USI i = 0; i < dim; i++) {
137  nnz += colId[i].size();
138  }
139 
140  A.row = dim;
141  A.col = dim;
142  A.nnz = nnz;
143 
144  // IA
145  OCP_USI count = 0;
146  A.IA[0] = 0;
147  for (OCP_USI i = 1; i < dim + 1; i++) {
148  USI nnz_Row = colId[i - 1].size();
149  A.IA[i] = A.IA[i - 1] + nnz_Row;
150 
151  for (USI j = 0; j < nnz_Row; j++) {
152  A.JA[count] = colId[i - 1][j];
153  A.val[count] = val[i - 1][j];
154  count++;
155  }

159 OCP_INT ScalarFaspSolver::Solve()
160 {
161  OCP_INT status = FASP_SUCCESS;
162 
163  const OCP_INT print_level = inParam.print_level;
164  const OCP_INT solver_type = inParam.solver_type;
165  const OCP_INT precond_type = inParam.precond_type;
166  const OCP_INT output_type = inParam.output_type;
167 
168  if (output_type) {
169  const char* outputfile = "out/Solver.out";
170  printf("Redirecting outputs to file: %s ...\n", outputfile);
171  freopen(outputfile, "w", stdout); // open a file for stdout
172  }
173 
174  // Preconditioned Krylov methods
175  if (solver_type >= 1 && solver_type <= 20) {
176 
177  // Using no preconditioner for Krylov iterative methods
178  if (precond_type == PREC_NULL) {
179  status = fasp_solver_dcsr_krylov(&A, &b, &x, &itParam);
180  }
181 
182  // Using diag(A) as preconditioner for Krylov iterative methods
183  else if (precond_type == PREC_DIAG) {
184  status = fasp_solver_dcsr_krylov_diag(&A, &b, &x, &itParam);
185  }
186 
187  // Using AMG as preconditioner for Krylov iterative methods
188  else if (precond_type == PREC_AMG || precond_type == PREC_FMG) {
189  if (print_level > PRINT_NONE) fasp_param_amg_print(&amgParam);
190  status = fasp_solver_dcsr_krylov_amg(&A, &b, &x, &itParam, &amgParam);
191  }
192 
193  // Using ILU as preconditioner for Krylov iterative methods
194  else if (precond_type == PREC_ILU) {
195  if (print_level > PRINT_NONE) fasp_param_ilu_print(&iluParam);
196  status = fasp_solver_dcsr_krylov_ilu(&A, &b, &x, &itParam, &iluParam);
197  }
198 
199  // Undefined iterative methods
200  else {
201  printf("### ERROR: Wrong preconditioner type %d!!!\n", precond_type);
202  status = ERROR_SOLVER_PRECTYPE;
203  }
204 
205  }
206 
207  // AMG as the iterative solver
208  else if (solver_type == SOLVER_AMG) {
209  if (print_level > PRINT_NONE) fasp_param_amg_print(&amgParam);
210  fasp_solver_amg(&A, &b, &x, &amgParam);
211  }
212 
213  // Full AMG as the iterative solver
214  else if (solver_type == SOLVER_FMG) {
215  if (print_level > PRINT_NONE) fasp_param_amg_print(&amgParam);
216  fasp_solver_famg(&A, &b, &x, &amgParam);
217  }
218 
219 #if WITH_MUMPS // use MUMPS directly
220  else if (solver_type == SOLVER_MUMPS) {
221  status = fasp_solver_mumps(&A, &b, &x, print_level);
222  if (status >= 0) status = 1; // Direct solver returns 1
223  }
224 #endif
225 
226 #if WITH_SuperLU // use SuperLU directly
227  else if (solver_type == SOLVER_SUPERLU) {
228  status = fasp_solver_superlu(&A, &b, &x, print_level);
229  if (status >= 0) status = 1; // Direct solver returns 1
230  }
231 #endif
232 
233 #if WITH_UMFPACK // use UMFPACK directly
234  else if (solver_type == SOLVER_UMFPACK) {
235  // Need to sort the matrix A for UMFPACK to work
236  dCSRmat A_trans = fasp_dcsr_create(A.row, A.col, A.nnz);
237  fasp_dcsr_transz(&A, NULL, &A_trans);
238  fasp_dcsr_sort(&A_trans);
239  status = fasp_solver_umfpack(&A_trans, &b, &x, print_level);
240  fasp_dcsr_free(&A_trans);
241  if (status >= 0) status = 1; // Direct solver returns 1
242  }
243 #endif
244 
245 #ifdef WITH_PARDISO // use PARDISO directly
246  else if (solver_type == SOLVER_PARDISO) {
247  fasp_dcsr_sort(&A);
248  status = fasp_solver_pardiso(&A, &b, &x, print_level);
249  if (status >= 0) status = 1; // Direct solver returns 1
250  }
251 #endif
252 
253  else {
254  printf("### ERROR: Wrong solver type %d!!!\n", solver_type);
255  status = ERROR_SOLVER_TYPE;
256  }
257 
258  if (status < 0) {
259  printf("### ERROR: Solver failed! Exit status = %d.\n\n", status);
260  }
261 
262  if (output_type) fclose(stdout);
263  return status;
264 }

350 void VectorFaspSolver::AssembleMat(const vector<vector<USI>>& colId,
351  const vector<vector<OCP_DBL>>& val,
352  const OCP_USI& dim,
353  const USI& blockDim,
354  vector<OCP_DBL>& rhs,
355  vector<OCP_DBL>& u)
356 {
357  OCP_USI nrow = dim * blockDim;
358  // b & x
359  b.row = nrow;
360  b.val = rhs.data();
361  x.row = nrow;
362  x.val = u.data(); // x will be set to zero later
363 
364  // fsc & order
365  fsc.row = nrow;
366  order.row = nrow;
367 
368  // nnz
369  OCP_USI nnz = 0;
370  for (OCP_USI i = 0; i < dim; i++) {
371  nnz += colId[i].size();
372  }
373 
374  // Asc
375  Asc.ROW = dim;
376  Asc.COL = dim;
377  Asc.nb = blockDim;
378  Asc.NNZ = nnz;
379 
380  // A
381  A.ROW = dim;
382  A.COL = dim;
383  A.nb = blockDim;
384  A.NNZ = nnz;
385 
386  OCP_USI count = 0;
387  OCP_USI count2 = 0;
388  OCP_USI size_row;
389  A.IA[0] = 0;
390  for (OCP_USI i = 1; i < dim + 1; i++) {
391  USI nnb_Row = colId[i - 1].size();
392  A.IA[i] = A.IA[i - 1] + nnb_Row;
393 
394  for (USI j = 0; j < nnb_Row; j++) {
395  A.JA[count] = colId[i - 1][j];
396  count++;
397  }
398  size_row = nnb_Row * blockDim * blockDim;
399  for (USI k = 0; k < size_row; k++) {
400  A.val[count2] = val[i - 1][k];
401  count2++;
402  }
403  }
404 
405 #ifdef DEBUG
406  // check x and b ---- for test
407  for (int i = 0; i < nrow; i++) {
408  if (!isfinite(b.val[i])) OCP_ABORT("vFasp b is infinite!");
409  if (!isfinite(x.val[i])) OCP_ABORT("vFasp x is infinite!");
410  }
411  // check A ---- for test
412  for (int i = 0; i < A.NNZ; i++) {
413  if (!isfinite(A.val[i])) OCP_ABORT("vFasp A is infinite!");
414  }
415 #endif // DEBUG
416 }
