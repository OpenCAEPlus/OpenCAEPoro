224 void BulkConn::AssembleMatIMPEC(LinearSystem& myLS,
225  const Bulk& myBulk,
226  const OCP_DBL& dt) const
227 {
228  OCP_FUNCNAME;
229  myLS.AddDim(numBulk);
230 
231  // accumulate term
232  OCP_DBL Vpp, Vp, vf, vfP, P;
233  for (OCP_USI n = 0; n < numBulk; n++) {
234  vf = myBulk.vf[n];
235  vfP = myBulk.vfP[n];
236  P = myBulk.lP[n];
237  Vpp = myBulk.rockVntg[n] * myBulk.poroP[n];
238  Vp = myBulk.rockVp[n];
239 
240  myLS.NewDiag(n, Vpp - vfP);
241  myLS.AddRhs(n, (Vpp - vfP) * P + dt * (vf - Vp));
242  }
243 
244  // flux term
245  OCP_USI bId, eId;
246  OCP_USI uId_np_j;
247  OCP_DBL valupi, valdowni;
248  OCP_DBL valup, rhsup, valdown, rhsdown;
249  OCP_DBL dD, tmp;
250 
251  const USI np = myBulk.numPhase;
252  const USI nc = myBulk.numCom;
253 
254  // Be careful when first bulk has no neighbors!
255  for (OCP_USI c = 0; c < numConn; c++) {
256  bId = iteratorConn[c].bId;
257  eId = iteratorConn[c].eId;
258  dD = GRAVITY_FACTOR * (myBulk.depth[bId] - myBulk.depth[eId]);
259 
260  valup = 0;
261  rhsup = 0;
262  valdown = 0;
263  rhsdown = 0;
264 
265  for (USI j = 0; j < np; j++) {
266  uId_np_j = upblock[c * np + j] * np + j;
267  if (!myBulk.phaseExist[uId_np_j]) continue;
268 
269  valupi = 0;
270  valdowni = 0;
271 
272  for (USI i = 0; i < nc; i++) {
273  valupi += myBulk.vfi[bId * nc + i] * myBulk.xij[uId_np_j * nc + i];
274  valdowni += myBulk.vfi[eId * nc + i] * myBulk.xij[uId_np_j * nc + i];
275  }
276 
277  tmp = myBulk.xi[uId_np_j] * upblock_Trans[c * np + j] * dt;
278  valup += tmp * valupi;
279  valdown += tmp * valdowni;
280  tmp *= upblock_Rho[c * np + j] * dD
281  - (myBulk.Pc[bId * np + j] - myBulk.Pc[eId * np + j]);
282  rhsup += tmp * valupi;
283  rhsdown -= tmp * valdowni;
284  }
285  myLS.AddDiag(bId, valup);
286  myLS.AddDiag(eId, valdown);
287  myLS.NewOffDiag(bId, eId, -valup);
288  myLS.NewOffDiag(eId, bId, -valdown);
289  myLS.AddRhs(bId, rhsup);
290  myLS.AddRhs(eId, rhsdown);
291  }
292 }

295 void BulkConn::CalFluxIMPEC(const Bulk& myBulk)
296 {
297  OCP_FUNCNAME;
298 
299  // calculate a step flux using iteratorConn
300  OCP_USI bId, eId, uId;
301  OCP_USI bId_np_j, eId_np_j;
302  OCP_BOOL exbegin, exend, exup;
303  OCP_DBL rho, dP, Akd;
304 
305  const USI np = myBulk.numPhase;
306 
307  for (OCP_USI c = 0; c < numConn; c++) {
308  bId = iteratorConn[c].bId;
309  eId = iteratorConn[c].eId;
310  Akd = CONV1 * CONV2 * iteratorConn[c].area;
311 
312  for (USI j = 0; j < np; j++) {
313  bId_np_j = bId * np + j;
314  eId_np_j = eId * np + j;
315 
316  exbegin = myBulk.phaseExist[bId_np_j];
317  exend = myBulk.phaseExist[eId_np_j];
318 
319  if ((exbegin) && (exend)) {
320  rho = (myBulk.rho[bId_np_j] + myBulk.rho[eId_np_j]) / 2;
321  } else if (exbegin && (!exend)) {
322  rho = myBulk.rho[bId_np_j];
323  } else if ((!exbegin) && (exend)) {
324  rho = myBulk.rho[eId_np_j];
325  } else {
326  upblock[c * np + j] = bId;
327  upblock_Trans[c * np + j] = 0;
328  upblock_Velocity[c * np + j] = 0;
329  continue;
330  }
331 
332  dP = (myBulk.Pj[bId_np_j] - GRAVITY_FACTOR * rho * myBulk.depth[bId])
333  - (myBulk.Pj[eId_np_j] - GRAVITY_FACTOR * rho * myBulk.depth[eId]);
334  if (dP < 0) {
335  uId = eId;
336  exup = exend;
337  }
338  else {
339  uId = bId;
340  exup = exbegin;
341  }
342 
343  upblock_Rho[c * np + j] = rho;
344  upblock[c * np + j] = uId;
345 
346  if (exup) {
347  upblock_Trans[c * np + j] = Akd * myBulk.kr[uId * np + j] / myBulk.mu[uId * np + j];
348  upblock_Velocity[c * np + j] = upblock_Trans[c * np + j] * dP;
349  } else {
350  upblock_Trans[c * np + j] = 0;
351  upblock_Velocity[c * np + j] = 0;
352  }
353  }
354  }

422 void BulkConn::AssembleMat_FIM(LinearSystem& myLS,
423  const Bulk& myBulk,
424  const OCP_DBL& dt) const
425 {
426  OCP_FUNCNAME;
427 
428  myLS.AddDim(numBulk);
429 
430  const USI np = myBulk.numPhase;
431  const USI nc = myBulk.numCom;
432  const USI ncol = nc + 1;
433  const USI ncol2 = np * nc + np;
434  const USI bsize = ncol * ncol;
435  const USI bsize2 = ncol * ncol2;
436 
437  vector<OCP_DBL> bmat(bsize, 0);
438 
439  // Accumulation term
440  for (USI i = 1; i < ncol; i++) {
441  bmat[i * ncol + i] = 1;
442  }
443  for (OCP_USI n = 0; n < numBulk; n++) {
444  bmat[0] = myBulk.rockVntg[n] * myBulk.poroP[n] - myBulk.vfP[n];
445  for (USI i = 0; i < nc; i++) {
446  bmat[i + 1] = -myBulk.vfi[n * nc + i];
447  }
448  myLS.NewDiag(n, bmat);
449  }
450 
451  // flux term
452  OCP_DBL Akd;
453  OCP_DBL transJ, transIJ;
454  vector<OCP_DBL> dFdXpB(bsize, 0);
455  vector<OCP_DBL> dFdXpE(bsize, 0);
456  vector<OCP_DBL> dFdXsB(bsize2, 0);
457  vector<OCP_DBL> dFdXsE(bsize2, 0);
458 
459  OCP_USI bId, eId, uId;
460  OCP_USI bId_np_j, eId_np_j, uId_np_j;
461  OCP_BOOL phaseExistBj, phaseExistEj;
462  OCP_DBL kr, mu, xi, xij, rhoP, xiP, muP, rhox, xix, mux;
463  OCP_DBL dP, dGamma;
464  OCP_DBL tmp;
465 
466  for (OCP_USI c = 0; c < numConn; c++) {
467  bId = iteratorConn[c].bId;
468  eId = iteratorConn[c].eId;
469  Akd = CONV1 * CONV2 * iteratorConn[c].area;
470  fill(dFdXpB.begin(), dFdXpB.end(), 0.0);
471  fill(dFdXpE.begin(), dFdXpE.end(), 0.0);
472  fill(dFdXsB.begin(), dFdXsB.end(), 0.0);
473  fill(dFdXsE.begin(), dFdXsE.end(), 0.0);
474  dGamma = GRAVITY_FACTOR * (myBulk.depth[bId] - myBulk.depth[eId]);
475 
476  for (USI j = 0; j < np; j++) {
477  uId = upblock[c * np + j];
478  bId_np_j = bId * np + j;
479  eId_np_j = eId * np + j;
480  uId_np_j = uId * np + j;
481 
482  if (!myBulk.phaseExist[uId_np_j]) continue;
483 
484  phaseExistBj = myBulk.phaseExist[bId_np_j];
485  phaseExistEj = myBulk.phaseExist[eId_np_j];
486 
487  dP = myBulk.Pj[bId_np_j] - myBulk.Pj[eId_np_j] -
488  upblock_Rho[c * np + j] * dGamma;
489  xi = myBulk.xi[uId_np_j];
490  kr = myBulk.kr[uId_np_j];
491  mu = myBulk.mu[uId_np_j];
492  muP = myBulk.muP[uId_np_j];
493  xiP = myBulk.xiP[uId_np_j];
494  rhoP = myBulk.rhoP[uId_np_j];
495  transJ = Akd * kr / mu;
496 
497  for (USI i = 0; i < nc; i++) {
498  xij = myBulk.xij[uId_np_j * nc + i];
499  transIJ = xij * xi * transJ;
500 
501  // Pressure -- Primary var
502  dFdXpB[(i + 1) * ncol] += transIJ;
503  dFdXpE[(i + 1) * ncol] -= transIJ;
504 
505  tmp = xij * transJ * xiP * dP;
506  tmp += -transIJ * muP / mu * dP;
507  if (!phaseExistEj) {
508  tmp += transIJ * (-rhoP * dGamma);
509  dFdXpB[(i + 1) * ncol] += tmp;
510  } else if (!phaseExistBj) {
511  tmp += transIJ * (-rhoP * dGamma);
512  dFdXpE[(i + 1) * ncol] += tmp;
513  } else {
514  dFdXpB[(i + 1) * ncol] +=
515  transIJ * (-myBulk.rhoP[bId_np_j] * dGamma) / 2;
516  dFdXpE[(i + 1) * ncol] +=
517  transIJ * (-myBulk.rhoP[eId_np_j] * dGamma) / 2;
518  if (bId == uId) {
519  dFdXpB[(i + 1) * ncol] += tmp;
520  } else {
521  dFdXpE[(i + 1) * ncol] += tmp;
522  }
523  }
524  // Second var
525  if (bId == uId) {
526  // Saturation
527  for (USI k = 0; k < np; k++) {
528  dFdXsB[(i + 1) * ncol2 + k] +=
529  transIJ * myBulk.dPcj_dS[bId_np_j * np + k];
530  tmp =
531  Akd * xij * xi / mu * myBulk.dKr_dS[uId_np_j * np + k] * dP;
532  dFdXsB[(i + 1) * ncol2 + k] += tmp;
533  dFdXsE[(i + 1) * ncol2 + k] -=
534  transIJ * myBulk.dPcj_dS[eId_np_j * np + k];
535  }
536  // Cij
537  if (!phaseExistEj) {
538  for (USI k = 0; k < nc; k++) {
539  rhox = myBulk.rhox[uId_np_j * nc + k];
540  xix = myBulk.xix[uId_np_j * nc + k];
541  mux = myBulk.mux[uId_np_j * nc + k];
542  tmp = -transIJ * rhox * dGamma;
543  tmp += xij * transJ * xix * dP;
544  tmp += -transIJ * mux / mu * dP;
545  dFdXsB[(i + 1) * ncol2 + np + j * nc + k] += tmp;
546  }
547  dFdXsB[(i + 1) * ncol2 + np + j * nc + i] += xi * transJ * dP;
548  } else {
549  for (USI k = 0; k < nc; k++) {
550  rhox = myBulk.rhox[bId_np_j * nc + k] / 2;
551  xix = myBulk.xix[uId_np_j * nc + k];
552  mux = myBulk.mux[uId_np_j * nc + k];
553  tmp = -transIJ * rhox * dGamma;
554  tmp += xij * transJ * xix * dP;
555  tmp += -transIJ * mux / mu * dP;
556  dFdXsB[(i + 1) * ncol2 + np + j * nc + k] += tmp;
557  dFdXsE[(i + 1) * ncol2 + np + j * nc + k] +=
558  -transIJ * myBulk.rhox[eId_np_j * nc + k] / 2 * dGamma;
559  }
560  dFdXsB[(i + 1) * ncol2 + np + j * nc + i] += xi * transJ * dP;
561  }
562  } else {
563  // Saturation
564  for (USI k = 0; k < np; k++) {
565  dFdXsB[(i + 1) * ncol2 + k] +=
566  transIJ * myBulk.dPcj_dS[bId_np_j * np + k];
567  dFdXsE[(i + 1) * ncol2 + k] -=
568  transIJ * myBulk.dPcj_dS[eId_np_j * np + k];
569  tmp =
570  Akd * xij * xi / mu * myBulk.dKr_dS[uId_np_j * np + k] * dP;
571  dFdXsE[(i + 1) * ncol2 + k] += tmp;
572  }
573  // Cij
574  if (!phaseExistBj) {
575  for (USI k = 0; k < nc; k++) {
576  rhox = myBulk.rhox[uId_np_j * nc + k];
577  xix = myBulk.xix[uId_np_j * nc + k];
578  mux = myBulk.mux[uId_np_j * nc + k];
579  tmp = -transIJ * rhox * dGamma;
580  tmp += xij * transJ * xix * dP;
581  tmp += -transIJ * mux / mu * dP;
