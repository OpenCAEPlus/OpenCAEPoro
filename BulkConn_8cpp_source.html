92 void BulkConn::CalAkd(const Bulk& myBulk)
93 {
94  OCP_USI bId, eId;
95  OCP_DBL areaB, areaE;
96  OCP_DBL T1, T2;
97  for (OCP_USI c = 0; c < numConn; c++) {
98  bId = iteratorConn[c].bId;
99  eId = iteratorConn[c].eId;
100  areaB = iteratorConn[c].areaB;
101  areaE = iteratorConn[c].areaE;
102  switch (iteratorConn[c].direction)
103  {
104  case 1:
105  T1 = myBulk.ntg[bId] * myBulk.rockKx[bId] * areaB;
106  T2 = myBulk.ntg[eId] * myBulk.rockKx[eId] * areaE;
107  break;
108  case 2:
109  T1 = myBulk.ntg[bId] * myBulk.rockKy[bId] * areaB;
110  T2 = myBulk.ntg[eId] * myBulk.rockKy[eId] * areaE;
111  break;
112  case 3:
113  T1 = myBulk.rockKz[bId] * areaB;
114  T2 = myBulk.rockKz[eId] * areaE;
115  break;
116  default:
117  OCP_ABORT("Wrong Direction!");
118  }
119  iteratorConn[c].area = 1 / (1 / T1 + 1 / T2);
120  }
121 }

174 void BulkConn::CalFluxFIMS(const Grid& myGrid, const Bulk& myBulk)
175 {
176  OCP_FUNCNAME;
177 
178  const USI np = myBulk.numPhase;
179  OCP_USI bId, eId, uId;
180  OCP_USI bId_np_j, eId_np_j;
181  OCP_DBL Pbegin, Pend, rho;
182 
183  for (OCP_USI c = 0; c < numConn; c++) {
184  bId = iteratorConn[c].bId;
185  eId = iteratorConn[c].eId;
186 
187  for (USI j = 0; j < np; j++) {
188  bId_np_j = bId * np + j;
189  eId_np_j = eId * np + j;
190 
191  OCP_BOOL exbegin = myBulk.phaseExist[bId_np_j];
192  OCP_BOOL exend = myBulk.phaseExist[eId_np_j];
193 
194  if ((exbegin) && (exend)) {
195  rho = (myBulk.rho[bId_np_j] * myBulk.S[bId_np_j] +
196  myBulk.rho[eId_np_j] * myBulk.S[eId_np_j]) /
197  (myBulk.S[bId_np_j] + myBulk.S[eId_np_j]);
198  } else if (exbegin && (!exend)) {
199  rho = myBulk.rho[bId_np_j];
200  } else if ((!exbegin) && (exend)) {
201  rho = myBulk.rho[eId_np_j];
202  } else {
203  upblock[c * np + j] = bId;
204  upblock_Rho[c * np + j] = 0;
205  continue;
206  }
207  Pbegin = myBulk.Pj[bId_np_j];
208  Pend = myBulk.Pj[eId_np_j];
209  uId = bId;
210  OCP_DBL dP = (Pbegin - GRAVITY_FACTOR * rho * myBulk.depth[bId]) -
211  (Pend - GRAVITY_FACTOR * rho * myBulk.depth[eId]);
212  if (dP < 0) {
213  uId = eId;
214  }
215  upblock[c * np + j] = uId;
216  upblock_Rho[c * np + j] = rho;
217  }
218  }

221 void BulkConn::CalResFIMS(vector<OCP_DBL>& res, const Bulk& myBulk, const OCP_DBL& dt)
222 {
223  OCP_FUNCNAME;
224 
225  const USI np = myBulk.numPhase;
226  const USI nc = myBulk.numCom;
227  const USI len = nc + 1;
228  OCP_USI bId, eId, uId, bIdb;
229 
230  // Accumalation Term
231  for (OCP_USI n = 0; n < numBulk; n++) {
232 
233  bId = n * len;
234  bIdb = n * nc;
235 
236  res[bId] = myBulk.rockVp[n] - myBulk.vf[n];
237  for (USI i = 0; i < nc; i++) {
238  res[bId + 1 + i] = myBulk.Ni[bIdb + i] - myBulk.lNi[bIdb + i];
239  }
240  }
241 
242  OCP_USI bId_np_j, eId_np_j, uId_np_j;
243  OCP_DBL Pbegin, Pend, rho, dP;
244  OCP_DBL tmp, dNi;
245  OCP_DBL Akd;
246  // Flux Term
247  // Calculate the upblock at the same time.
248  for (OCP_USI c = 0; c < numConn; c++) {
249  bId = iteratorConn[c].bId;
250  eId = iteratorConn[c].eId;
251  Akd = CONV1 * CONV2 * iteratorConn[c].area;
252 
253  for (USI j = 0; j < np; j++) {
254  bId_np_j = bId * np + j;
255  eId_np_j = eId * np + j;
256 
257  OCP_BOOL exbegin = myBulk.phaseExist[bId_np_j];
258  OCP_BOOL exend = myBulk.phaseExist[eId_np_j];
259 
260  if ((exbegin) && (exend)) {
261  rho = (myBulk.rho[bId_np_j] * myBulk.S[bId_np_j] +
262  myBulk.rho[eId_np_j] * myBulk.S[eId_np_j]) /
263  (myBulk.S[bId_np_j] + myBulk.S[eId_np_j]);
264  } else if (exbegin && (!exend)) {
265  rho = myBulk.rho[bId_np_j];
266  } else if ((!exbegin) && (exend)) {
267  rho = myBulk.rho[eId_np_j];
268  } else {
269  upblock[c * np + j] = bId;
270  upblock_Rho[c * np + j] = 0;
271  continue;
272  }
273  Pbegin = myBulk.Pj[bId_np_j];
274  Pend = myBulk.Pj[eId_np_j];
275  uId = bId;
276  dP = (Pbegin - GRAVITY_FACTOR * rho * myBulk.depth[bId]) -
277  (Pend - GRAVITY_FACTOR * rho * myBulk.depth[eId]);
278  if (dP < 0) {
279  uId = eId;
280  }
281  upblock_Rho[c * np + j] = rho;
282  upblock[c * np + j] = uId;
283 
284  uId_np_j = uId * np + j;
285  if (!myBulk.phaseExist[uId_np_j]) continue;
286  tmp = dt * Akd * myBulk.xi[uId_np_j] * myBulk.kr[uId_np_j] /
287  myBulk.mu[uId_np_j] * dP;
288 
289  for (USI i = 0; i < nc; i++) {
290  dNi = tmp * myBulk.xij[uId_np_j * nc + i];
291  res[bId * len + 1 + i] += dNi;
292  res[eId * len + 1 + i] -= dNi;
293  }
294  }
