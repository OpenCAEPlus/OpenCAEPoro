145 Grid::CalAkdOrthogonalGrid(const OCP_USI& bId, const OCP_USI& eId, const USI& direction)
146 {
147  OCP_DBL T1;
148  OCP_DBL T2;
149  switch (direction) {
150  case 1:
151  // x-direction
152  T1 = kx[bId] * ntg[bId] * dy[bId] * dz[bId] / dx[bId];
153  T2 = kx[eId] * ntg[eId] * dy[eId] * dz[eId] / dx[eId];
154  return (2 / (1 / T1 + 1 / T2));
155  break;
156  case 2:
157  // y-direction
158  T1 = ky[bId] * ntg[bId] * dz[bId] * dx[bId] / dy[bId];
159  T2 = ky[eId] * ntg[eId] * dz[eId] * dx[eId] / dy[eId];
160  return (2 / (1 / T1 + 1 / T2));
161  break;
162  case 3:
163  // z-direction -- no ntg
164  T1 = kz[bId] * dx[bId] * dy[bId] / dz[bId];
165  T2 = kz[eId] * dx[eId] * dy[eId] / dz[eId];
166  return (2 / (1 / T1 + 1 / T2));
167  break;
168  default:
169  OCP_ABORT("Wrong Direction!");
170  }

173 void Grid::CalDepthVOrthogonalGrid()
174 {
175  depth.resize(numGrid, 0);
176  OCP_USI nxny = nx * ny;
177  // 0th layer
178  for (USI j = 0; j < ny; j++) {
179  for (USI i = 0; i < nx; i++) {
180  OCP_USI id = j * nx + i;
181  depth[id] = tops[id] + dz[id] / 2;
182  }
183  }
184  // 1th - (nz-1)th layer
185  for (USI k = 1; k < nz; k++) {
186  OCP_USI knxny = k * nxny;
187  for (USI j = 0; j < ny; j++) {
188  for (USI i = 0; i < nx; i++) {
189  OCP_USI id = knxny + j * nx + i;
190  depth[id] = depth[id - nxny] + dz[id - nxny] / 2 + dz[id] / 2;
191  }
192  }
193  }
194 
195  v.resize(numGrid);
196  for (OCP_USI i = 0; i < numGrid; i++) v[i] = dx[i] * dy[i] * dz[i];
197 }

253 OCP_DBL Grid::CalAkdCornerGrid(const GeneralConnect& conn)
254 {
255  OCP_USI bId = conn.begin;
256  OCP_USI eId = conn.end;
257  OCP_DBL bArea = conn.Ad_dd_begin;
258  OCP_DBL eArea = conn.Ad_dd_end;
259  OCP_DBL T1, T2;
260 
261  switch (conn.directionType) {
262  case 1:
263  T1 = ntg[bId] * kx[bId] * bArea;
264  T2 = ntg[eId] * kx[eId] * eArea;
265  return 1 / (1 / T1 + 1 / T2);
266  break;
267  case 2:
268  T1 = ntg[bId] * ky[bId] * bArea;
269  T2 = ntg[eId] * ky[eId] * eArea;
270  return 1 / (1 / T1 + 1 / T2);
271  break;
272  case 3:
273  T1 = kz[bId] * bArea;
274  T2 = kz[eId] * eArea;
275  return 1 / (1 / T1 + 1 / T2);
276  break;
277  default:
278  OCP_ABORT("Wrong Direction Type!");
279  }
