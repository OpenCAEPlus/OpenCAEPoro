86  return false;

98  dt /= 2;

100  cout << "CFL is too big" << endl;
101  return false;
102  }
103 
104  rs.MassConseveIMPEC(dt);
105 
106  // third check: Ni check

108  dt /= 2;

110  cout << "Negative Ni occurs\n";
111  return false;
112  }
113 
114  rs.CalVpore();
115  rs.CalFlashIMPEC();
116 
117  // fouth check: Volume error check
