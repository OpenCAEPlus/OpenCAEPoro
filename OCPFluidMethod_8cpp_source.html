76 OCP_BOOL OCP_IMPEC::UpdateProperty(Reservoir& rs, OCPControl& ctrl)
77 {
78  OCP_DBL& dt = ctrl.current_dt;
79 
80  // first check : Pressure check
81  OCP_INT flagCheck = rs.CheckP(OCP_TRUE, OCP_TRUE);
82  switch (flagCheck) {
83  case 1:
84  // Negative Bulk P, well P, or Perforation P
85  dt /= 2;
86  return OCP_FALSE;
87  case 2:
88  // Switch Well opt Mode, or close the cross-flow perforation
89  dt /= 1;
90  return OCP_FALSE;
91  default:
92  // All right
93  break;
94  }
95 
96  // Calculate Flux between bulks and between bulks and wells
97  rs.CalFLuxIMPEC();
98 
99  // second check : CFL check
100  OCP_DBL cfl = rs.CalCFL(dt);
101  if (cfl > 1) {
102  dt /= 2;
103  rs.ResetVal01IMPEC();
104  cout << "CFL is too big" << endl;
105  return OCP_FALSE;
106  }
107 
108  rs.MassConserveIMPEC(dt);
109 
110  // third check: Ni check
111  if (!rs.CheckNi()) {
112  dt /= 2;
113  rs.ResetVal02IMPEC();
114  cout << "Negative Ni occurs\n";
115  return OCP_FALSE;
116  }
117 
118  rs.CalRock();
119  rs.CalFlashIMPEC();
120 
121  // fouth check: Volume error check
122  if (!rs.CheckVe(0.01)) {
123  // cout << ctrl.GetCurTime() << "Days" << "=======" << endl;
124  dt /= 2;
125  rs.ResetVal03IMPEC();
126  return OCP_FALSE;
127  }
128 
129  rs.CalKrPc();
130  rs.CalConnFluxIMPEC();
131  // rs.allWells.ShowWellStatus(rs.bulk);
132 
133  return OCP_TRUE;
134 }

671 OCP_BOOL OCP_AIMc::FinishNR(Reservoir& rs, OCPControl& ctrl)
672 {
673  OCP_USI dSn;
674  const OCP_DBL NRdSmax = rs.GetNRdSmax(dSn);
675  const OCP_DBL NRdPmax = rs.GetNRdPmax();
676  // const OCP_DBL NRdNmax = rs.GetNRdNmax();
677 
678 #ifdef DEBUG
679  cout << "### DEBUG: Residuals = " << setprecision(3) << scientific
680  << resFIM.maxRelRes0_v << " " << resFIM.maxRelRes_v << " "
681  << resFIM.maxRelRes_mol << " " << NRdPmax << " " << NRdSmax << endl;
682 #endif
683 
684  if (ctrl.iterNR > ctrl.ctrlNR.maxNRiter) {
685  ctrl.current_dt *= ctrl.ctrlTime.cutFacNR;
686  rs.ResetFIM();
687  rs.CalResAIMc(resFIM, ctrl.current_dt);
688  resFIM.maxRelRes0_v = resFIM.maxRelRes_v;
689  ctrl.ResetIterNRLS();
690  cout << "### WARNING: NR not fully converged! Cut time step size and repeat! "
691  "current dt = "
692  << fixed << setprecision(3) << ctrl.current_dt << " days\n";
693  return OCP_FALSE;
694  }
695 
696  if (((resFIM.maxRelRes_v <= resFIM.maxRelRes0_v * ctrl.ctrlNR.NRtol ||
697  resFIM.maxRelRes_v <= ctrl.ctrlNR.NRtol ||
698  resFIM.maxRelRes_mol <= ctrl.ctrlNR.NRtol) &&
699  resFIM.maxWellRelRes_mol <= ctrl.ctrlNR.NRtol) ||
700  (fabs(NRdPmax) <= ctrl.ctrlNR.NRdPmin &&
701  fabs(NRdSmax) <= ctrl.ctrlNR.NRdSmin)) {
702 
703  OCP_INT flagCheck = rs.CheckP(OCP_FALSE, OCP_TRUE);
704 #if DEBUG
705  if (flagCheck > 0) {
706  cout << ">> Switch well constraint: Case " << flagCheck << endl;
707  }
708 #endif
709 
710  switch (flagCheck) {
711  case 1:
712  ctrl.current_dt *= ctrl.ctrlTime.cutFacNR;
713  rs.ResetFIM();
714  rs.CalResAIMc(resFIM, ctrl.current_dt);
715  resFIM.maxRelRes0_v = resFIM.maxRelRes_v;
716  ctrl.ResetIterNRLS();
717  return OCP_FALSE;
718  case 2:
719  // ctrl.current_dt /= 1;
720  rs.ResetFIM();
721  rs.CalResAIMc(resFIM, ctrl.current_dt);
722  resFIM.maxRelRes0_v = resFIM.maxRelRes_v;
723  ctrl.ResetIterNRLS();
724  return OCP_FALSE;
725  default:
726  // Update IMPEC Bulk Properties
727  rs.CalFlashAIMc01();
728  rs.CalKrPcAIMc();
729  return OCP_TRUE;
730  break;
731  }
732  } else {
733  return OCP_FALSE;
734  }
