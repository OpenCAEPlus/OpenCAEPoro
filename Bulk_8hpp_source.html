61  flagSkip.resize(numBulk);
62  PSkip.resize(numBulk);
63  TSkip.resize(numBulk);
64  minEigenSkip.resize(numBulk);

67  OCP_BOOL IfSkip(const OCP_DBL& Pin, const OCP_DBL& Tin, const OCP_DBL& Ntin,
68  const OCP_DBL* Niin, const OCP_USI& n, const USI& numCom_1) const {
69  if (flagSkip[n]) {
70  if (fabs(1 - PSkip[n] / Pin) >= minEigenSkip[n] / 10) {
71  return OCP_FALSE;
72  }
73  OCP_DBL Nt_w = Ntin - Niin[numCom_1];
74  for (USI i = 0; i < numCom_1; i++) {
75  if (fabs(Niin[i] / Nt_w - ziSkip[n*numCom_1 + i]) >= minEigenSkip[n] / 10) {
76  return OCP_FALSE;
77  }
78  }
79  if (fabs(TSkip[n] - Tin) >= minEigenSkip[n] * 10) {
80  return OCP_FALSE;
81  }
82  return OCP_TRUE;
83  }
84  else {
85  return OCP_FALSE;
86  }
87  }
