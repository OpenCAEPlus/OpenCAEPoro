96 void Well::CalWI_Peaceman(const Bulk& myBulk)

99

101  // especially the dz
102 
103  for (USI p = 0; p < numPerf; p++) {
104  if (perf[p].WI > 0) {
105  break;
106  } else {
107  const OCP_USI Idb = perf[p].location;
108  const OCP_DBL dx = myBulk.dx[Idb];
109  const OCP_DBL dy = myBulk.dy[Idb];
110  const OCP_DBL dz = myBulk.dz[Idb] * myBulk.ntg[Idb];
111  OCP_DBL ro = 0;
112  switch (perf[p].direction) {
113  case X_DIRECTION:
114  {
115  const OCP_DBL kykz = myBulk.rockKy[Idb] * myBulk.rockKz[Idb];
116  const OCP_DBL ky_kz = myBulk.rockKy[Idb] / myBulk.rockKz[Idb];
117  assert(kykz > 0);
118  ro = 0.28 * pow((dy * dy * pow(1 / ky_kz, 0.5) +
119  dz * dz * pow(ky_kz, 0.5)),
120  0.5);
121  ro /= (pow(ky_kz, 0.25) + pow(1 / ky_kz, 0.25));
122 
123  if (perf[p].kh < 0) {
124  perf[p].kh = (dx * pow(kykz, 0.5));
125  }
126  break;
127  }
128  case Y_DIRECTION:
129  {
130  const OCP_DBL kzkx = myBulk.rockKz[Idb] * myBulk.rockKx[Idb];
131  const OCP_DBL kz_kx = myBulk.rockKz[Idb] / myBulk.rockKx[Idb];
132  assert(kzkx > 0);
133  ro = 0.28 * pow((dz * dz * pow(1 / kz_kx, 0.5) +
134  dx * dx * pow(kz_kx, 0.5)),
135  0.5);
136  ro /= (pow(kz_kx, 0.25) + pow(1 / kz_kx, 0.25));
137 
138  if (perf[p].kh < 0) {
139  perf[p].kh = (dy * pow(kzkx, 0.5));
140  }
141  break;
142  }
143  case Z_DIRECTION:
144  {
145  const OCP_DBL kxky = myBulk.rockKx[Idb] * myBulk.rockKy[Idb];
146  const OCP_DBL kx_ky = myBulk.rockKx[Idb] / myBulk.rockKy[Idb];
147  assert(kxky > 0);
148  ro = 0.28 * pow((dx * dx * pow(1 / kx_ky, 0.5) +
149  dy * dy * pow(kx_ky, 0.5)),
150  0.5);
151  ro /= (pow(kx_ky, 0.25) + pow(1 / kx_ky, 0.25));
152

155  }
156  break;
157  }

160  }
