493 void T_FIM::CalThermalConduct(BulkConn& conn, Bulk& bk) const
494 {
495  const OCP_USI nb = bk.numBulk;
496  const OCP_USI np = bk.numPhase;
497 
498  for (OCP_USI n = 0; n < nb; n++) {
499  if (bk.bType[n] > 0) {
500  // fluid bulk
501  OCP_DBL tmp = 0;
502  for (USI j = 0; j < np; j++) {
503  tmp += bk.S[n * np + j] * bk.thconp[j];
504  bk.ktS[n * np + j] = bk.poro[n] * bk.thconp[j];
505  }
506  bk.kt[n] = bk.poro[n] * tmp + (1 - bk.poro[n]) * bk.thconr[n];
507  bk.ktP[n] = bk.poroP[n] * (tmp - bk.thconr[n]);
508  bk.ktT[n] = bk.poroT[n] * (tmp - bk.thconr[n]);
509  }
510  else {
511  // non fluid bulk
512  bk.kt[n] = bk.thconr[n];
513  }
514  }
515 
516  OCP_USI bId, eId;
517  OCP_DBL areaB, areaE, T1, T2;
518  OCP_DBL tmpB, tmpE;
519 
520  for (OCP_USI c = 0; c < conn.numConn; c++) {
521  bId = conn.iteratorConn[c].BId();
522  eId = conn.iteratorConn[c].EId();
523  if (bk.bType[bId] > 0 && bk.bType[eId] > 0) {
524  // fluid bulk connections
525  areaB = conn.iteratorConn[c].AreaB();
526  areaE = conn.iteratorConn[c].AreaE();
527  T1 = bk.kt[bId] * areaB;
528  T2 = bk.kt[eId] * areaE;
529  conn.Adkt[c] = 1 / (1 / T1 + 1 / T2);
530 
531  tmpB = pow(conn.Adkt[c], 2) / pow(T1, 2) * areaB;
532  tmpE = pow(conn.Adkt[c], 2) / pow(T2, 2) * areaE;
533  conn.AdktP[c * 2 + 0] = tmpB * bk.ktP[bId];
534  conn.AdktP[c * 2 + 1] = tmpE * bk.ktP[eId];
535  conn.AdktT[c * 2 + 0] = tmpB * bk.ktT[bId];
536  conn.AdktT[c * 2 + 1] = tmpE * bk.ktT[eId];

712 void T_FIM::CalRes(Reservoir& rs, const OCP_DBL& t, const OCP_DBL& dt, const OCP_BOOL& resetRes0)
713 {
714  const Bulk& bk = rs.bulk;
715  const USI nb = bk.numBulk;
716  const USI np = bk.numPhase;
717  const USI nc = bk.numCom;
718  const USI len = nc + 2;
719  OCPRes& Res = bk.res;
720  BulkConn& conn = rs.conn;
721 
722  Res.SetZero();
723 
724  // Bulk to Bulk
725 
726  OCP_USI bId, eId, uId, bIdb;
727  // Accumalation Term
728  for (OCP_USI n = 0; n < nb; n++) {
729  if (bk.bType[n] > 0) {
730  // Fluid bulk
731  bId = n * len;
732  bIdb = n * nc;
733  // Volume Conservation
734  Res.resAbs[bId] = bk.rockVp[n] - bk.vf[n];
735  // Mass Conservation
736  for (USI i = 0; i < nc; i++) {
737  Res.resAbs[n * len + 1 + i] = bk.Ni[bIdb + i] - bk.lNi[bIdb + i];
738  }
739  // Energy Conservation
740  Res.resAbs[n * len + nc + 1] = (bk.vf[n] * bk.Uf[n] + bk.vr[n] * bk.Hr[n])
741  - (bk.lvf[n] * bk.lUf[n] + bk.lvr[n] * bk.lHr[n]);
742  }
743  else {
744  // Non fluid bulk
745  Res.resAbs[n * len + nc + 1] = bk.vr[n] * bk.Hr[n] - bk.lvr[n] * bk.lHr[n];
746  }
747 
748  // Heat Loss
749  if (bk.hLoss.IfHeatLoss() && bk.bLocation[n] > 0) {
750  const OCP_DBL lambda = bk.bLocation[n] == 1 ? bk.hLoss.obD : bk.hLoss.ubD;
751  const OCP_DBL kappa = bk.bLocation[n] == 1 ? bk.hLoss.obK : bk.hLoss.ubK;
752  // dT
753  Res.resAbs[n * len + nc + 1] += dt * kappa * (2*(bk.T[n] - bk.initT[n]) / sqrt(lambda * t) - bk.hLoss.p[n])
754  * bk.dx[n] * bk.dy[n];
755  }
756  }
757 
758  OCP_USI bId_np_j, eId_np_j, uId_np_j;
759  OCP_DBL rho, dP, dT, dNi, Akd;
760 
761  for (OCP_USI c = 0; c < conn.numConn; c++) {
762  bId = conn.iteratorConn[c].BId();
763  eId = conn.iteratorConn[c].EId();
764 
765  // Thermal conductive term all exist
766  dT = bk.T[bId] - bk.T[eId];
767  Res.resAbs[bId * len + 1 + nc] += conn.Adkt[c] * dT * dt;
768  Res.resAbs[eId * len + 1 + nc] -= conn.Adkt[c] * dT * dt;
769 
770  if (bk.bType[bId] > 0 && bk.bType[eId] > 0) {
771  // Fluid Bulk Connection
772  Akd = CONV1 * CONV2 * conn.iteratorConn[c].Area();
773 
774  for (USI j = 0; j < np; j++) {
775  bId_np_j = bId * np + j;
776  eId_np_j = eId * np + j;
777 
778  const OCP_BOOL exbegin = bk.phaseExist[bId_np_j];
779  const OCP_BOOL exend = bk.phaseExist[eId_np_j];
780 
781  if ((exbegin) && (exend)) {
782  rho = (bk.rho[bId_np_j] + bk.rho[eId_np_j]) / 2;
783  }
784  else if (exbegin && (!exend)) {
785  rho = bk.rho[bId_np_j];
786  }
787  else if ((!exbegin) && (exend)) {
788  rho = bk.rho[eId_np_j];
789  }
790  else {
791  conn.upblock[c * np + j] = bId;
792  conn.upblock_Rho[c * np + j] = 0;
793  continue;
794  }
795 
796  uId = bId;
797  dP = (bk.Pj[bId_np_j] - GRAVITY_FACTOR * rho * bk.depth[bId])
798  - (bk.Pj[eId_np_j] - GRAVITY_FACTOR * rho * bk.depth[eId]);
799  if (dP < 0) {
800  uId = eId;
801  }
802  conn.upblock_Rho[c * np + j] = rho;
803  conn.upblock[c * np + j] = uId;
804  uId_np_j = uId * np + j;
805 
806  if (bk.phaseExist[uId_np_j]) {
807  conn.upblock_Velocity[c * np + j] =
808  Akd * bk.kr[uId_np_j] / bk.mu[uId_np_j] * dP;
809  }
810  else {
811  conn.upblock_Velocity[c * np + j] = 0;
812  continue;
813  }
814 
815  const OCP_DBL tmpV = dt * conn.upblock_Velocity[c * np + j]
816  * bk.xi[uId_np_j];
817 
818  for (USI i = 0; i < nc; i++) {
819  dNi = tmpV * bk.xij[uId_np_j * nc + i];
820  Res.resAbs[bId * len + 1 + i] += dNi;
821  Res.resAbs[eId * len + 1 + i] -= dNi;
822  }
823  Res.resAbs[bId * len + 1 + nc] += tmpV * bk.H[uId_np_j];
824  Res.resAbs[eId * len + 1 + nc] -= tmpV * bk.H[uId_np_j];
825  }
826  }
827  }
828 
829  // Well to Bulk
830  USI wId = nb * len;
831  for (const auto& wl : rs.allWells.wells) {
832  if (wl.IsOpen()) {
833  // Well to Bulk
834 
835  for (USI p = 0; p < wl.PerfNum(); p++) {
836  const OCP_USI k = wl.PerfLocation(p);
837  // Mass Conservation
838  for (USI i = 0; i < nc; i++) {
839  Res.resAbs[k * len + 1 + i] += wl.PerfQi_lbmol(p, i) * dt;
840  }
841  }
842 
843  if (wl.WellType() == INJ) {
844  // Energy Conservation -- Well to Bulk
845  const OCP_DBL Hw = bk.flashCal[0]->CalInjWellEnthalpy(wl.InjTemp(), &wl.InjZi()[0]);
846  for (USI p = 0; p < wl.PerfNum(); p++) {
847  const OCP_USI k = wl.PerfLocation(p);
848  Res.resAbs[k * len + 1 + nc] += wl.PerfInjQt_ft3(p) * wl.PerfXi(p) * Hw * dt;
849  }
850 
851  // Well Self -- Injection
852  switch (wl.OptMode()) {
853  case BHP_MODE:
854  Res.resAbs[wId] = wl.BHP() - wl.MaxBHP();
855  break;
856  case RATE_MODE:
857  case ORATE_MODE:
858  case GRATE_MODE:
859  case WRATE_MODE:
860  case LRATE_MODE:
861  Res.resAbs[wId] = wl.MaxRate();
862  for (USI i = 0; i < nc; i++) {
863  Res.resAbs[wId] += wl.Qi_lbmol(i);
864  }
865  //if (opt.reInj) {
866  // for (auto& w : opt.connWell) {
867  // OCP_DBL tmp = 0;
868  // for (USI i = 0; i < numCom; i++) {
869  // tmp += allWell[w].qi_lbmol[i];
870  // }
871  // tmp *= opt.reInjFactor;
872  // Res.resAbs[bId] += tmp;
873  // }
874  //}
875  Res.maxWellRelRes_mol =
876  max(Res.maxWellRelRes_mol, fabs(Res.resAbs[wId] / wl.MaxRate()));
877  break;
878  default:
879  OCP_ABORT("Wrong well opt mode!");
880  break;
881  }
882  }
883  else {
884  // Energy Conservation -- Bulk to Well
885  for (USI p = 0; p < wl.PerfNum(); p++) {
886  const OCP_USI k = wl.PerfLocation(p);
887  // Mass Conservation
888  for (USI j = 0; j < np; j++) {
889  Res.resAbs[k * len + 1 + nc] += wl.PerfProdQj_ft3(p, j) * bk.xi[k * np + j] * bk.H[k * np + j] * dt;
890  }
891  }
892 
893  // Well Self -- Production
894  switch (wl.OptMode()) {
895  case BHP_MODE:
896  Res.resAbs[wId] = wl.BHP() - wl.MinBHP();
897  break;
898  case RATE_MODE:
899  case ORATE_MODE:
900  case GRATE_MODE:
901  case WRATE_MODE:
902  case LRATE_MODE:
903  wl.CalProdWeight(bk);
904  Res.resAbs[wId] = -wl.MaxRate();
905  for (USI i = 0; i < nc; i++) {
906  Res.resAbs[wId] += wl.Qi_lbmol(i) * wl.ProdWeight(i);
907  }
908  Res.maxWellRelRes_mol =
909  max(Res.maxWellRelRes_mol, fabs(Res.resAbs[wId] / wl.MaxRate()));
910  break;
911  default:
912  OCP_ABORT("Wrong well opt mode!");
913  break;
914  }
915  }
916  wId += len;
917  }
918  }
919 
920  // Calculate RelRes
921  OCP_DBL tmp;
922  for (OCP_USI n = 0; n < nb; n++) {
923 
924  // Energy equations always exist
925  OCP_DBL eT = bk.vf[n] * bk.Uf[n] + bk.vr[n] * bk.Hr[n];
926  tmp = fabs(Res.resAbs[n * len + nc + 1] / eT);
927  if (Res.maxRelRes_E < tmp) {
928  Res.maxRelRes_E = tmp;
929  Res.maxId_E = n;
930  }
931 
932  if (bk.bType[n] > 0) {
933  // Fluid Bulk
934  for (USI i = 0; i < len - 1; i++) {
935  tmp = fabs(Res.resAbs[n * len + i] / bk.rockVp[n]);
936  if (Res.maxRelRes_V < tmp) {
937  Res.maxRelRes_V = tmp;
938  Res.maxId_V = n;
939  }
940  }
941 
942  for (USI i = 1; i < len - 1; i++) {
943  tmp = fabs(Res.resAbs[n * len + i] / bk.Nt[n]);
944  if (Res.maxRelRes_N < tmp) {
945  Res.maxRelRes_N = tmp;
946  Res.maxId_N = n;
947  }
948  }
949  }
950  }
951 
952  Dscalar(Res.resAbs.size(), -1.0, Res.resAbs.data());
953  if (resetRes0) Res.SetInitRes();
954 }
