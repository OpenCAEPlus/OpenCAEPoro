93void OCPControl::CalNextTstepIMPEC(const Reservoir& reservoir)
94{
95 last_dt = current_dt;
96 current_time += current_dt;
97
98 OCP_DBL c1, c2, c3, c4, c;
99 c1 = c2 = c3 = c4 = 10;
100
101 OCP_DBL dPmax = reservoir.bulk.GetdPmax();
102 OCP_DBL dNmax = reservoir.bulk.GetdNmax();
103 OCP_DBL dSmax = reservoir.bulk.GetdSmax();
104 OCP_DBL dVmax = reservoir.bulk.GetdVmax();
105
106 if (dPmax > TINY) c1 = ctrlPreTime.dPlim / dPmax;
107 if (dSmax > TINY) c2 = ctrlPreTime.dSlim / dSmax;
108 if (dNmax > TINY) c3 = ctrlPreTime.dNlim / dNmax;
109 if (dVmax > TINY) c4 = ctrlPreTime.dVlim / dVmax;
110
111 c = min(min(c1, c2), min(c3, c4));
112 c = max(ctrlTime.minChopFac, c);
113 c = min(ctrlTime.maxIncreFac, c);
114
115 current_dt *= c;
116
117 if (current_dt > ctrlTime.timeMax) current_dt = ctrlTime.timeMax;
118 if (current_dt < ctrlTime.timeMin) current_dt = ctrlTime.timeMin;
119
120 OCP_DBL dt = end_time - current_time;
121 if (current_dt > dt) current_dt = dt;
122}
