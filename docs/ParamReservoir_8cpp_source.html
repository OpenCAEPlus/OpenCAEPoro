152void ParamReservoir::setVal(vector<T>& obj, const T& val, const vector<USI>& index)
153{
154 USI Nx = dimens.nx;
155 USI Ny = dimens.ny;
156 OCP_USI NxNy = Nx * Ny;
157 OCP_USI id = 0;
158
159 for (USI k = index[4]; k <= index[5]; k++) {
160 for (USI j = index[2]; j <= index[3]; j++) {
161 for (USI i = index[0]; i <= index[1]; i++) {
162 id = k * NxNy + j * Nx + i;
163 obj[id] = val;
164 }
165 }
166 }

241void ParamReservoir::InputEQUALS(ifstream& ifs)
242{

244 vector<string> vbuf;
245
246 while (ReadLine(ifs, vbuf)) {

251 index[4] = 0, index[5] = dimens.nz - 1;

253 string objName = vbuf[0];
254 OCP_DBL val = stod(vbuf[1]);
255
256 DealDefault(vbuf);
257

263

266 objPtr->resize(dimens.nx * dimens.ny);
267 index[4] = index[5] = 0;
268 } else {
269 objPtr->resize(numGrid);
270 }
271 setVal(*objPtr, val, index);
272 } else {
273 ParamCheck("Wrong Var Name: " + objName);
274 exit(0);
275 }
276 }

388void ParamReservoir::InputTABLE(ifstream& ifs, const string& tabName)
389{
390 TableSet* obj;
391 obj = FindPtr_T(tabName);
392 if (obj == nullptr) {
393 ParamCheck("Wrong Tab Name :" + tabName);
394 exit(0);
395 }
396
397 USI col = obj->colNum;
398 vector<vector<OCP_DBL>> tmpTab(col);
399
400 vector<string> vbuf;
401 while (ReadLine(ifs, vbuf)) {
402 if (vbuf[0] == "/") break;
403
404 for (USI i = 0; i < col; i++) {
405 tmpTab[i].push_back(stod(vbuf[i]));
406 }
407
408 if (vbuf.back() == "/") {
409 obj->data.push_back(tmpTab);
410 for (USI j = 0; j < col; j++) {
411 tmpTab[j].clear();
412 }
413 }
414 }
415 if (!tmpTab[0].empty()) obj->data.push_back(tmpTab);
416
417 obj->DisplayTable();
418 std::cout << "TABLE" << endl;

497void ParamReservoir::InputRegion(ifstream& ifs, const string& keyword)
498{
499 Type_A_r<OCP_DBL>* ptr = &PVTNUM;
500 USI lim = NTPVT;
501
502 if (keyword == "SATNUM") {
503 ptr = &SATNUM;
504 lim = NTSFUN;
505 }
506
507 ptr->activity = true;
508 ptr->data.reserve(numGrid);
509 vector<string> vbuf;
510 vector<OCP_USI> obj;
511 vector<USI> region;
512
513 while (ReadLine(ifs, vbuf)) {
514 if (vbuf[0] == "/") break;
515
516 DealData(vbuf, obj, region);
517
518#ifdef DEBUG
519 // check region
520 for (auto r : region) {
521 if (r > lim) {
522 ParamCheck("Region is out of Range!");
523 exit(0);
524 }
525 }
526#endif // DEBUG
527
528 USI len = obj.size();
529 for (USI i = 0; i < len; i++) {
530 for (OCP_USI j = 0; j < obj[i]; j++) {
531 ptr->data.push_back(region[i]);
532 }
533 }
534 }
535
536 cout << "Number of tables = " << lim << endl;
537 cout << &SATNUM << endl << &PVTNUM << endl;
538}
