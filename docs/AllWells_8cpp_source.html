19void AllWells::InputParam(const ParamWell& paramWell)
20{
21 OCP_FUNCNAME;
22
23 USI len = paramWell.solSet.size();
24 solvents.resize(len);
25 for (USI i = 0; i < len; i++) {
26 solvents[i] = paramWell.solSet[i];
27 }
28
29 numWell = paramWell.well.size();
30 wells.resize(numWell);
31 USI t = paramWell.criticalTime.size();
32 vector<USI> wellCriticalTime;
33 for (USI w = 0; w < numWell; w++) {
34 wells[w].name = paramWell.well[w].name;
35 wells[w].group = paramWell.well[w].group;
36 wells[w].depth = paramWell.well[w].depth;
37 wells[w].I = paramWell.well[w].I - 1;
38 wells[w].J = paramWell.well[w].J - 1;
39 wells[w].InputPerfo(paramWell.well[w]);
40
41 // opt
42 wells[w].optSet.resize(t);
43 USI n = paramWell.well[w].optParam.size();
44 wellCriticalTime.clear();
45 wellCriticalTime.resize(n + 1);
46 for (USI i = 0; i < n; i++) {
47 wellCriticalTime[i] = paramWell.well[w].optParam[i].d;
48 }
49 wellCriticalTime.back() = t;
50 for (USI i = 0; i < n; i++) {
51 for (USI d = wellCriticalTime[i]; d < wellCriticalTime[i + 1]; d++) {
52 wells[w].optSet[d] = WellOpt(paramWell.well[w].optParam[i].opt);
53 }
54 }
55 }

270void AllWells::CalReInjFluid(const Bulk& myBulk)
271{
272 for (auto& wG : wellGroup) {
273 if (wG.reInj) {
274 const USI nc = myBulk.GetComNum();
275 // const USI np = myBulk.GetPhaseNum();
276 fill(wG.zi.begin(), wG.zi.end(), 0.0);
277 for (auto& prod : wellGroup[wG.prodGroup].wIdPROD) {
278 Daxpy(nc, 1.0, &wells[prod].qi_lbmol[0], &wG.zi[0]);
279 }
280 OCP_DBL qt = Dnorm1(nc, &wG.zi[0]);
281 if (fabs(qt) < 1E-8) {
282 // Recalculate zi
283 fill(wG.zi.begin(), wG.zi.end(), 0.0);
284 for (auto& prod : wellGroup[wG.prodGroup].wIdPROD) {
285 wells[prod].CalReInjFluid(myBulk, wG.zi);
286 }
287 qt = Dnorm1(nc, &wG.zi[0]);
288 }
289 flashCal[0]->Flash(PRESSURE_STD, TEMPERATURE_STD, &wG.zi[0], 0, 0, 0);
290 Dcopy(nc, &wG.zi[0], &flashCal[0]->xij[wG.injPhase * nc]);
291 wG.xi = flashCal[0]->xi[wG.injPhase];
292 wG.factor = wG.xi * flashCal[0]->v[wG.injPhase] / qt;
293 // assign to every open injection well in wG
294 for (auto& w : wG.wIdINJ) {
295 if (wells[w].WellState()) {
296 wells[w].opt.reInj = true;
297 wells[w].opt.connWell = wG.wIdPROD;
298 wells[w].opt.injPhase = wG.injPhase;
299 wells[w].opt.zi = wG.zi;
300 wells[w].opt.xiINJ = wG.xi;
301 wells[w].opt.factor = wG.factor;
302 wells[w].opt.maxRate = -wG.saleRate * wG.xi * 1000; // Mscf -> ft3 -> lbmol
303 }
304 }
305 //for (USI i = 0; i < nc; i++) {
306 // cout << scientific << wG.zi[i] << " ";
307 //}
308 //cout << "xi: " << wG.xi << " Factor: " << wG.factor;
309 //cout << endl;
310 }

337OCP_INT AllWells::CheckP(const Bulk& myBulk)
338{
339 OCP_FUNCNAME;
340
341 // 0 : All correct
342 // 1 : negative P, cut the timestep and resolve
343 // 2.1 : change well mode to BHP, resolve
344 // 2.2 : crossflow happens, then close corresponding perf, resolve
345 bool flag2 = false;
346 bool flag3 = false;
347
348 for (USI w = 0; w < numWell; w++) {
349 if (wells[w].WellState()) {
350
351 OCP_INT flag = wells[w].CheckP(myBulk);
352//#ifdef _DEBUG
353 // wells[w].ShowPerfStatus();
354//#endif // _DEBUG
355 switch (flag) {
356 case 1:
357 return 1;
358 case 2:
359 flag2 = true;
360 break;
361 case 3:
362 flag3 = true;
363 break;
364 default:
365 break;
366 }
367 }
368 }
369
370 if (flag2 || flag3) return 2;
371
372 return 0;
373}
