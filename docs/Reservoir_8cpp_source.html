438void Reservoir::PrintSolFIM(const string& outfile) const
439{
440 ofstream outu(outfile);
441 if (!outu.is_open()) cout << "Can not open " << outfile << endl;
442 const OCP_USI nb = bulk.numBulk;
443 const OCP_USI nc = bulk.numCom;
444
445 for (OCP_USI n = 0; n < nb; n++) {
446 // Pressure
447 outu << bulk.P[n] << "\n";
448 // Ni
449 for (USI i = 0; i < nc; i++) {
450 outu << bulk.Ni[n * nc + i] << "\n";
451 }
452 }
453 // Well Pressure
454 for (USI w = 0; w < wellgroup.numWell; w++) {
455 outu << wellgroup.GetWBHP(w) << "\n";
456 }
457 outu.close();
458}
