53 void HeatLoss::CalHeatLoss(const vector<USI>& location, const vector<OCP_DBL>& T, const vector<OCP_DBL>& lT,
54  const vector<OCP_DBL>& initT, const OCP_DBL& t, const OCP_DBL& dt)
55 {
56  if (ifHLoss) {
57  OCP_DBL lambda, d, dT, theta;
58  OCP_DBL tmp, q;
59  for (OCP_USI n = 0; n < numBulk; n++) {
60  if (location[n] > 0) {
61  // overburden or underburden
62  lambda = location[n] == 1 ? obD : ubD;
63 
64  dT = T[n] - lT[n];
65  theta = T[n] - initT[n];
66  d = sqrt(lambda * t) / 2;
67  tmp = 3 * pow(d, 2) + lambda * dt;
68  p[n] = (theta * (lambda * dt / d) + lI[n] - dT * (pow(d, 3) / (lambda * dt))) / tmp;
69  pT[n] = (lambda * dt / d - pow(d, 3) / (lambda * dt)) / tmp;
70  q = (2 * p[n] * d - theta + pow(d, 2) * dT / (lambda * dt)) / (2 * pow(d, 2));
71  I[n] = theta * d + p[n] * pow(d, 2) + 2 * q * pow(d, 3);
72  }
73  }
74  }
