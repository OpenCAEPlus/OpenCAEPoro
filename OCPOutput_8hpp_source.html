249  for (OCP_USI n = 0; n < numGrid; n++) {

251  if (n % (nx * ny) == 0) myRPT << "\n\n";
252 
253  if (n % nx == 0) {
254  GetIJKGrid(I, J, K, n);
255  myRPT << GetIJKformat("*", to_string(J), to_string(K), IJKspace);
256  }
257 
258  if (gbPair[n].IsAct()) {
259  bId = gbPair[n].GetId();
260  myRPT << setw(10) << fixed << setprecision(3)
261  << gridVal[bId * gap] * alpha;
262  } else {
263  myRPT << setw(10) << " --- ";
264  }
265  }
266  } else {
267  for (OCP_USI n = 0; n < numGrid; n++) {
268  if (n % nx == 0) myRPT << "\n";
269  if (n % (nx * ny) == 0) myRPT << "\n\n";
270 
271  if (n % nx == 0) {
272  GetIJKGrid(I, J, K, n);
273  myRPT << GetIJKformat("*", to_string(J), to_string(K), IJKspace);
274  }
275  myRPT << setw(10) << fixed << setprecision(3) << gridVal[n * gap] * alpha;
276  }
277  }
278 
279  myRPT << "\n\n\n";
