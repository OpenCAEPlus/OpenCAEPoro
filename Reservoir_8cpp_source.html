79 OCP_DBL Reservoir::CalCFL(const OCP_DBL& dt) const
80 {
81  fill(bulk.cfl.begin(), bulk.cfl.end(), 0.0);
82  const USI np = bulk.numPhase;
83 
84  for (OCP_USI c = 0; c < conn.numConn; c++) {
85  for (USI j = 0; j < np; j++) {
86  const OCP_USI uId = conn.upblock[c * np + j];
87 
88  if (bulk.phaseExist[uId * np + j]) {
89  bulk.cfl[uId * np + j] += fabs(conn.upblock_Velocity[c * np + j]) * dt;
90  }
91  }
92  }
93 
94  for (const auto& wl : allWells.wells) {
95  if (wl.IsOpen() && wl.WellType() == PROD) {
96  for (USI p = 0; p < wl.PerfNum(); p++) {
97  if (wl.PerfState(p) == OPEN) {
98  const OCP_USI k = wl.PerfLocation(p);
99 
100  for (USI j = 0; j < np; j++) {
101  bulk.cfl[k * np + j] += fabs(wl.PerfProdQj_ft3(p,j)) * dt;
102  }
103  }
104  }
105  }
106  }
107 
108  bulk.maxCFL = 0;
109  const OCP_USI len = bulk.numBulk * np;
110  for (OCP_USI n = 0; n < len; n++) {
111  if (bulk.phaseExist[n]) {
112  bulk.cfl[n] /= bulk.vj[n];
113 #ifdef DEBUG

115  OCP_ABORT("cfl is nan!");
116  }
117 #endif // DEBUG

119  }
