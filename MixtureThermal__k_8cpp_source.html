267

270  const OCP_DBL xiwT = xi[1] * (-ct1[1] - ct2[1] * dT + cpt[1] * dP);

274 void MixtureThermal_K01::FlashFIM(const OCP_DBL& Pin,
275  const OCP_DBL& Tin,
276  const OCP_DBL* Niin,
277  const OCP_DBL* Sjin,
278  const USI& lastNP,
279  const OCP_DBL* xijin,
280  const OCP_USI& bId)
281 {
282  // Assign value
283  P = Pin; T = Tin + CONV5;
284  const OCP_DBL dP = P - Pref;
285  const OCP_DBL dT = T - Tref;
286  Ni[0] = Niin[0];
287  Ni[1] = Niin[1];
288  Nt = Ni[0] + Ni[1];
289 
290  // phase viscosity
291  if (useViscTab) {
292  visc.Eval_All(0, T, data, cdata);
293  mu[0] = data[1];
294  mu[1] = data[2];
295  // d mu / dT
296  muT[0] = cdata[1];
297  muT[1] = cdata[2];
298 
299  }
300  else {
301  mu[0] = avisc[0] * exp(bvisc[0] / T);
302  mu[1] = avisc[1] * exp(bvisc[1] / T);
303  // d mu / dT
304  muT[0] = mu[0] * (-bvisc[0] / (T * T));
305  muT[1] = mu[1] * (-bvisc[1] / (T * T));
306  }
307 
308 
309  // phase molar density
310  xi[0] = xi_ref[0] * exp(cp[0] * dP - ct1[0] * dT -
311  ct2[0] * pow(dT, 2) / 2 + cpt[0] * dP * dT);
312  xi[1] = xi_ref[1] * exp(cp[1] * dP - ct1[1] * dT -
313  ct2[1] * pow(dT, 2) / 2 + cpt[1] * dP * dT);
314  // d xi / dP
315  xiP[0] = xi[0] * (cp[0] + cpt[0] * dT);
316  xiP[1] = xi[1] * (cp[1] + cpt[1] * dT);
317  // d xi / dT
318  xiT[0] = xi[0] * (-ct1[0] - ct2[0] * dT + cpt[0] * dP);
319  xiT[1] = xi[1] * (-ct1[1] - ct2[1] * dT + cpt[1] * dP);
320 
321  // phase mass density
322  rho[0] = MWp[0] * xi[0];
323  rho[1] = MWp[1] * xi[1];
324  // d rho / d P
325  rhoP[0] = MWp[0] * xiP[0];
326  rhoP[1] = MWp[1] * xiP[1];
327  // d rho / d T
328  rhoT[0] = MWp[0] * xiT[0];
329  rhoT[1] = MWp[1] * xiT[1];
330 
331  // phase volume
332  vj[0] = Ni[0] / xi[0];
333  vj[1] = Ni[1] / xi[1];
334  // total volume
335  vf = vj[0] + vj[1];
336 
337  // phase saturation
338  S[0] = vj[0] / vf;
339  S[1] = vj[1] / vf;
340 
341  // d vf/ d Ni
342  vfi[0] = 1 / xi[0];
343  vfi[1] = 1 / xi[1];
344  // d vf / d P
345  vfP = -(vj[0] * xiP[0] / xi[0] + vj[1] * xiP[1] / xi[1]);
346  // d vf / d T
347  vfT = -(vj[0] * xiT[0] / xi[0] + vj[1] * xiT[1] / xi[1]);
348 
349  // Derivative of secondary vars with respect to primary vars
350  dXsdXp[0] = (-vj[0] * xiP[0] / xi[0] - S[0] * vfP) / vf; // dSo / dP
351  dXsdXp[1] = (1 / xi[0] - S[0] * vfi[0]) / vf; // dSo / dNo
352  dXsdXp[2] = -S[0] * vfi[1] / vf; // dSo / dNw
353  dXsdXp[3] = (-vj[0] * xiT[0] / xi[0] - S[0] * vfT) / vf; // dSo / dT
354 
355  dXsdXp[4] = -dXsdXp[0]; // dSw / dP
356  dXsdXp[5] = -dXsdXp[1]; // dSw / dNo
357  dXsdXp[6] = -dXsdXp[2]; // dSw / dNw
358  dXsdXp[7] = -dXsdXp[3]; // dSw / dT
359 
360  CalEnthalpy();
361 }

363 void MixtureThermal_K01::CalEnthalpy()
364 {
365  fill(H.begin(), H.end(), 0.0);
366  fill(HT.begin(), HT.end(), 0.0);
367 
368  const OCP_DBL dT = T - Tref;
369 
370  if (liquid_based || simple_hvap) {
371  for (USI j = 0; j < numPhase; j++) {
372  for (USI i = 0; i < numCom; i++) {
373 
374  Hx[j * numCom + i] = (cpl1[i] * dT + 1.0 / 2 * cpl2[i] * pow(dT, 2) + 1.0 / 3 * cpl3[i] * pow(dT, 3) + 1.0 / 4 * cpl4[i] * pow(dT, 4));
375 
376  H[j] += xij[j * numCom + i] * Hx[j * numCom + i];
377 
378  HT[j] += xij[j * numCom + i] * (cpl1[i] + cpl2[i] * dT + cpl3[i] * pow(dT, 2) + cpl4[i] * pow(dT, 3));
379  }
380  }
381  }
382  else if (gas_based) {
383  for (USI j = 0; j < numPhase; j++) {
384  for (USI i = 0; i < numCom; i++) {
385  Hx[j * numCom + i] = cpg1[i] * dT
386  + 1.0 / 2 * cpg2[i] * pow(dT, 2)
387  + 1.0 / 3 * cpg3[i] * pow(dT, 3)
388  + 1.0 / 4 * cpg4[i] * pow(dT, 4);
389  H[j] += xij[j * numCom + i] * Hx[j * numCom + i];
390  HT[j] += xij[j * numCom + i] * (cpg1[i] + cpg2[i] * dT +
391  cpg3[i] * pow(dT, 2) + cpg4[i] * pow(dT, 3));
392 
393  if (T < Tcrit[i]) {
394  Hx[j * numCom + i] -= hvr[i] * pow((Tcrit[i] - T), ev[i]);
395 
396  H[j] -= xij[j * numCom + i] * hvr[i] * pow((Tcrit[i] - T), ev[i]);
397 
398  HT[j] += xij[j * numCom + i] * hvr[i] * ev[i] * pow((Tcrit[i] - T), ev[i] - 1);
399  }
400  }
401  }
402  }
403  else {
404  OCP_ABORT("WRONG Type !");
405  }
406 
407 
408  // Internal energy per unit volume of fluid
409 
410  // Uf, d Uf / d T, d Uf / d P
411  Uf = -P / (GRAVITY_FACTOR * CONV6);
412  UfP = -1 / (GRAVITY_FACTOR * CONV6);
413  UfT = 0;
414 
415  for (USI j = 0; j < numPhase; j++) {
416  // Uf
417  Uf += S[j] * xi[j] * H[j];
418  // dUf / dP
419  UfP += -(vj[j] * xiP[j] / xi[j] + S[j] * vfP) / vf * xi[j] * H[j];
420  UfP += xiP[j] * S[j] * H[j];
421  // dUf / dT
422  UfT += -(vj[j] * xiT[j] / xi[j] + S[j] * vfT) / vf * xi[j] * H[j];
423  UfT += (xiT[j] * H[j] + HT[j] * xi[j]) * S[j];
424  }
425 
426  // d Uf / d Ni
427  Ufi[0] = dXsdXp[1] * xi[0] * H[0] + dXsdXp[5] * xi[1] * H[1];
428  Ufi[1] = dXsdXp[2] * xi[0] * H[0] + dXsdXp[6] * xi[1] * H[1];
429 }

504  const vector<OCP_DBL>& prodPhase, vector<OCP_DBL>& prodWeight)
505 {
506  P = Pin; T = Tin + 460;

508  const OCP_DBL dT = T - Tref;
509  Ni[0] = Niin[0];
510  Ni[1] = Niin[1];
511  Nt = Ni[0] + Ni[1];
512 
513  // phase molar density
514  xi[0] = xi_ref[0] * exp(cp[0] * dP - ct1[0] * dT -
515  ct2[0] / 2 * pow(dT, 2) + cpt[0] * dP * dT);
516  xi[1] = xi_ref[1] * exp(cp[1] * dP - ct1[1] * dT -
517  ct2[1] / 2 * pow(dT, 2) + cpt[1] * dP * dT);

521  vj[1] = Ni[1] / xi[1];

526  factor[1] = vj[1] / qt / CONV1; // stb / lbmol
527 
528  OCP_DBL tmp = 0;
529  for (USI i = 0; i < 2; i++) {

531  }

534  }

540  vector<OCP_DBL>& prodRate)
541 {
542  // assign Value
543  P = Pin; T = Tin + 460;
544  const OCP_DBL dP = P - Pref;
545  const OCP_DBL dT = T - Tref;
546  Ni[0] = Niin[0];
547  Ni[1] = Niin[1];
548 
549  // phase molar density
550  xi[0] = xi_ref[0] * exp(cp[0] * dP - ct1[0] * dT -
551  ct2[0] / 2 * pow(dT, 2) + cpt[0] * dP * dT);
552  xi[1] = xi_ref[1] * exp(cp[1] * dP - ct1[1] * dT -
553  ct2[1] / 2 * pow(dT, 2) + cpt[1] * dP * dT);
554 
555  // phase volume
556  vj[0] = Ni[0] / xi[0];
557  vj[1] = Ni[1] / xi[1];
558 
559  prodRate[0] = vj[0] / CONV1; // stb
560  prodRate[1] = vj[1] / CONV1; // stb
561 }

564 void MixtureThermal_K01::SetupWellOpt(WellOpt& opt,
565  const vector<SolventINJ>& sols,
566  const OCP_DBL& Psurf,
567  const OCP_DBL& Tsurf)
568 {
569  const USI wellType = opt.WellType();
570  if (wellType == INJ) {
571  const string fluidName = opt.InjFluidType();
572  vector<OCP_DBL> tmpZi(numCom, 0);
573  if (fluidName == "WAT") {
574  tmpZi.back() = 1;
575  opt.SetInjProdPhase(WATER);
576  // lbmol / ft3 -> lbmol / bbl for
577  // injfluid Use flash in Bulk in surface condition
578  OCP_DBL tmp = CONV1 * XiPhase(Psurf, Tsurf, &tmpZi[0], WATER);
579  opt.SetInjFactor(tmp);
580  }
581  else {
582  OCP_ABORT("WRONG Fluid Type!");
583  }
584  opt.SetInjZi(tmpZi);
585  }
586  else if (wellType == PROD) {
587  vector<OCP_DBL> tmpWght(numPhase, 0);
588  switch (opt.OptMode()) {
589  case ORATE_MODE:
590  tmpWght[0] = 1;
591  break;
592  case WRATE_MODE:
593  tmpWght[1] = 1;
594  break;
595  case LRATE_MODE:
596  tmpWght[0] = 1;
597  tmpWght[1] = 1;
598  break;
599  default:
600  OCP_ABORT("WRONG optMode");
601  break;
602  }
603  opt.SetProdPhaseWeight(tmpWght);
604  }
605  else {
606  OCP_ABORT("Wrong Well Type!");
607  }
608 }
