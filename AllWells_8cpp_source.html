286 void AllWells::CalReInjFluid(const Bulk& myBulk)
287 {
288  for (auto& wG : wellGroup) {
289  if (wG.reInj) {
290  const USI nc = myBulk.GetComNum();
291  // const USI np = myBulk.GetPhaseNum();
292  fill(wG.reInjZi.begin(), wG.reInjZi.end(), 0.0);
293  for (auto& prod : wellGroup[wG.prodGroup].wIdPROD) {
294  Daxpy(nc, 1.0, &wells[prod].qi_lbmol[0], &wG.reInjZi[0]);
295  }
296  OCP_DBL qt = Dnorm1(nc, &wG.reInjZi[0]);
297  if (fabs(qt) < 1E-8) {
298  // Recalculate zi
299  fill(wG.reInjZi.begin(), wG.reInjZi.end(), 0.0);
300  for (auto& prod : wellGroup[wG.prodGroup].wIdPROD) {
301  wells[prod].CalReInjFluid(myBulk, wG.reInjZi);
302  }
303  qt = Dnorm1(nc, &wG.reInjZi[0]);
304  }
305  flashCal[0]->Flash(Psurf, Tsurf, &wG.reInjZi[0]);
306  Dcopy(nc, &wG.reInjZi[0], &flashCal[0]->xij[wG.reInjPhase * nc]);
307  wG.reInjXi = flashCal[0]->xi[wG.reInjPhase];
308  wG.reInjFactor = wG.reInjXi * flashCal[0]->vj[wG.reInjPhase] / qt;
309  // assign to every open injection well in wG
310  for (auto& w : wG.wIdINJ) {
311  if (wells[w].IsOpen()) {
312  wells[w].opt.reInj = OCP_TRUE;
313  wells[w].opt.connWell = wG.wIdPROD;
314  wells[w].opt.reInjPhase = wG.reInjPhase;
315  wells[w].opt.injZi = wG.reInjZi;
316  wells[w].opt.factorINJ = wG.reInjXi * 1000; // Blackoil Model prohibited
317  wells[w].opt.reInjFactor = wG.reInjFactor;
318  wells[w].opt.maxRate =
319  -wG.saleRate * wells[w].opt.factorINJ; // Mscf -> ft3 -> lbmol
320  }
321  }
322  }
323  }
