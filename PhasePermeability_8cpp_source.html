52 OCP_BOOL Miscible::CalFkFp(const OCP_USI& n, OCP_DBL& fk, OCP_DBL& fp)
53 {
54  if (surTen[n] >= surTenRef || surTen[n] <= TINY) {
55  Fk[n] = 1;
56  Fp[n] = 1;

59  else {
60  Fk[n] = fk = min(1.0, pow(surTen[n] / surTenRef, Fkexp));
